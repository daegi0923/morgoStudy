from collections import deque
def check(lst):
    if not lst:
        return False
    l = len(lst)
    if l == 1:
        return True
    visited = [0] * N
    start = lst[0]
    q = deque()
    q.append(start)
    while q:
        curr = q.popleft()
        visited[curr] = 1
        for i in range(N):
            if i in lst and R[curr][i] and not visited[i]:
                q.append(i)

    if sum(visited) == l:
        return True
    else:
